export async function getPage(userId: string): Promise<Page> {
  // Check circuit breaker before doing anything
  const breaker = getCaptchaBreaker(userId);
  if (breaker.isOpen()) {
    throw new CircuitBreakerOpenError(userId, 'captcha');
  }

  const page = await pool.getPage(userId);

  // Try to restore cookies if this is a fresh context
  const cookies = await loadCookies(userId);
  if (cookies && cookies.length > 0) {
    await page.context().addCookies(cookies);
    log.info({ userId, cookieCount: cookies.length }, 'Cookies restored');

    // Navigate to check if session is valid
    await page.goto('https://www.immobilienscout24.de/meinkonto/dashboard', {
      waitUntil: 'domcontentloaded',
    });

    // Check for CAPTCHA
    if (await detectCaptcha(page, userId)) {
      throw new CircuitBreakerOpenError(userId, 'captcha');
    }

    // Check if cookies gave us a valid session
    if (await isLoggedIn(page)) {
      log.info({ userId }, 'Session restored from cookies');
      breaker.recordSuccess();
      return page;
    }

    log.info({ userId }, 'Cookies expired — falling back to login');
  }

  // Need to login
  const user = await getUser(userId);
  let password: string;
  try {
    password = decrypt(user.immoscout_password_encrypted);
  } catch (err) {
    log.error({ userId }, 'Failed to decrypt password — credentials may be corrupted');
    throw err;
  }
  await login(page, user.immoscout_email, password);

  // Save new cookies after successful login
  const newCookies = await page.context().cookies();
  await saveCookies(userId, newCookies);

  breaker.recordSuccess();
  log.info({ userId }, 'Login complete, cookies saved');
  return page;
}

export async function isSessionHealthy(userId: string): Promise<boolean> {
  const breaker = getCaptchaBreaker(userId);

The patch introduces a few functional regressions in the core listing-monitor pipeline: overlapping users can lose auto-apply jobs, downstream failures can permanently drop listings, and only the first saved search is processed. Those issues make the new monitor behavior unreliable in normal usage.

Full review comments:

- [P1] Avoid persisting listings before the application/job is durable — /Users/david/Desktop/apartment-applicator/src/modules/listing-monitor/index.ts:50-70
  If `applications.insert(...)` or the subsequent `autoApplyQueue.add(...)` fails for a new expose, the listing row has already been inserted and future monitor cycles will treat that expose as a duplicate. In that scenario the auto-apply flow is never retried, so a transient DB/Redis failure permanently drops the listing for this user (and, because dedup is global on `immoscout_id`, potentially for every user).

  `jobId` is currently only `apply-${listing.immoscoutId}`. BullMQ deduplicates on `jobId` per queue, so if two users see the same ImmoScout listing, the second `add()` call will reuse/skip the first user's job instead of enqueueing a separate application for that user. This breaks multi-user operation whenever saved searches overlap.
